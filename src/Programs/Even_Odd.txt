//Program to check Given Number is Even and Odd.
#include <bits/stdc++.h>
using namespace std;
int main() {
  int n;
//Enter the Number
  cin >> n;
  if (n % 2 == 0)//Condition to Check Even Number.
    cout << "Entered Number is Even Number\n";
  else//Condition for Odd.
    cout<<"Entered Number is Odd Number\n";
}
